const screenEl = document.getElementById('screen');
const historyEl = document.getElementById('history');
const keysEl = document.getElementById('keys');

const langEl = document.getElementById('lang');
const titleEl = document.getElementById('title');
const hintEl = document.getElementById('hint');
const langLabelEl = document.getElementById('langLabel');
const tipTextEl = document.getElementById('tipText');

const MAX_LEN = 18;

const i18n = {
  en: {
    title: 'Calculator',
    hint: 'Keyboard: 0–9, + − × ÷, Enter, Backspace, Esc, .',
    language: 'Language',
    tip: 'Tip: Use x or X for ×, / for ÷.',
    error: 'Error'
  },
  hi: {
    title: 'कैलकुलेटर',
    hint: 'कीबोर्ड: 0–9, + − × ÷, Enter, Backspace, Esc, .',
    language: 'भाषा',
    tip: 'टिप: × के लिए x या X, ÷ के लिए / इस्तेमाल करें।',
    error: 'त्रुटि'
  }
};

const state = {
  lang: 'en',
  display: '0',
  a: null,
  op: null,            // '+', '-', '*', '/'
  waitingForB: false,
  justEvaluated: false
};

function symbol(op) {
  if (op === '*') return '×';
  if (op === '/') return '÷';
  return op;
}

function toHindiDigits(str) {
  const map = { '0':'०','1':'१','2':'२','3':'३','4':'४','5':'५','6':'६','7':'७','8':'८','9':'९' };
  return String(str).replace(/[0-9]/g, d => map[d]);
}

function localizeText(str) {
  return state.lang === 'hi' ? toHindiDigits(str) : String(str);
}

function formatForScreen(strOrNum) {
  let s = String(strOrNum);

  if (s === 'Infinity' || s === '-Infinity' || s === 'NaN') return i18n[state.lang].error;

  const n = Number(s);
  if (Number.isFinite(n)) {
    const abs = Math.abs(n);
    if ((abs !== 0 && (abs >= 1e12 || abs < 1e-6)) || s.length > MAX_LEN) {
      s = n.toExponential(8).replace(/\.?0+e/, 'e');
    } else if (s.length > MAX_LEN) {
      s = n.toPrecision(12).replace(/\.?0+$/, '');
    }
  }

  if (s.length > MAX_LEN) s = s.slice(0, MAX_LEN);
  return localizeText(s);
}

function setDisplay(value) {
  state.display = String(value);
  screenEl.textContent = formatForScreen(state.display);
}

function setHistory(text) {
  historyEl.textContent = text ? localizeText(text) : '';
}

function currentNumber() {
  const n = Number(state.display);
  return Number.isFinite(n) ? n : NaN;
}

function clearAll() {
  state.display = '0';
  state.a = null;
  state.op = null;
  state.waitingForB = false;
  state.justEvaluated = false;
  setHistory('');
  setDisplay('0');
}

function clearEntry() {
  state.display = '0';
  state.waitingForB = false;
  state.justEvaluated = false;
  setDisplay('0');
}

function backspace() {
  if (state.justEvaluated) state.justEvaluated = false;

  if (state.waitingForB) {
    state.waitingForB = false;
    state.display = '0';
  }

  if (state.display.length <= 1 || (state.display.length === 2 && state.display.startsWith('-'))) {
    setDisplay('0');
    return;
  }
  setDisplay(state.display.slice(0, -1));
}

function inputDigit(d) {
  if (state.justEvaluated) {
    state.a = null;
    state.op = null;
    state.waitingForB = false;
    state.justEvaluated = false;
    setHistory('');
    setDisplay('0');
  }

  if (state.waitingForB) {
    state.waitingForB = false;
    state.display = '0';
  }

  if (state.display === '0') setDisplay(d);
  else if (state.display.length < 32) setDisplay(state.display + d);
}

function inputDot() {
  if (state.justEvaluated) {
    state.a = null;
    state.op = null;
    state.waitingForB = false;
    state.justEvaluated = false;
    setHistory('');
    setDisplay('0');
  }

  if (state.waitingForB) {
    state.waitingForB = false;
    state.display = '0';
  }

  if (!state.display.includes('.')) setDisplay(state.display + '.');
}

function toggleSign() {
  if (state.display === '0' || state.display === '0.') return;
  if (state.display.startsWith('-')) setDisplay(state.display.slice(1));
  else setDisplay('-' + state.display);
}

function compute(a, op, b) {
  switch (op) {
    case '+': return a + b;
    case '-': return a - b;
    case '*': return a * b;
    case '/': return b === 0 ? NaN : a / b;
    default: return b;
  }
}

function chooseOp(op) {
  const b = currentNumber();

  if (state.op && state.a !== null && !state.waitingForB) {
    const result = compute(state.a, state.op, b);
    setHistory(`${formatPlain(state.a)} ${symbol(state.op)} ${formatPlain(b)}`);
    if (!Number.isFinite(result)) {
      setDisplay('NaN');
      state.a = null;
      state.op = null;
      state.waitingForB = true;
      state.justEvaluated = true;
      return;
    }
    state.a = result;
    setDisplay(String(result));
  } else if (state.a === null) {
    state.a = b;
  }

  state.op = op;
  state.waitingForB = true;
  state.justEvaluated = false;

  setHistory(`${formatPlain(state.a)} ${symbol(op)}`);
}

function equals() {
  if (!state.op || state.a === null) return;

  const b = currentNumber();
  const result = compute(state.a, state.op, b);

  setHistory(`${formatPlain(state.a)} ${symbol(state.op)} ${formatPlain(b)} =`);

  if (!Number.isFinite(result)) setDisplay('NaN');
  else setDisplay(String(result));

  state.a = Number.isFinite(result) ? result : null;
  state.op = null;
  state.waitingForB = true;
  state.justEvaluated = true;
}

function formatPlain(n) {
  // plain (non-localized) formatting for internal history string, then localized at render time
  const s = String(n);
  if (s === 'Infinity' || s === '-Infinity' || s === 'NaN') return i18n[state.lang].error;
  return s;
}

/* UI */
function applyLanguage(lang) {
  state.lang = lang;

  const t = i18n[state.lang];
  document.documentElement.lang = state.lang === 'hi' ? 'hi' : 'en';

  titleEl.textContent = t.title;
  hintEl.textContent = t.hint;
  langLabelEl.textContent = t.language;
  tipTextEl.textContent = t.tip;

  // Re-render current display/history in selected numeral system
  setDisplay(state.display);
  // historyEl already contains localized text; rebuild from state if you want exactness:
  // We'll keep it simple: leave as-is unless you want strict rebuild.
  // But we can safely re-localize by reading a non-localized source; we don't store it.
  // So: no-op for history beyond current content.
}

function handleKeysClick(e) {
  const btn = e.target.closest('button');
  if (!btn) return;

  const digit = btn.getAttribute('data-digit');
  const op = btn.getAttribute('data-op');
  const action = btn.getAttribute('data-action');

  if (digit !== null) inputDigit(digit);
  else if (op) chooseOp(op);
  else if (action === 'dot') inputDot();
  else if (action === 'equals') equals();
  else if (action === 'clear') clearAll();
  else if (action === 'ce') clearEntry();
  else if (action === 'backspace') backspace();
  else if (action === 'sign') toggleSign();
}

keysEl.addEventListener('click', handleKeysClick);

langEl.addEventListener('change', () => {
  applyLanguage(langEl.value);
});

// Keyboard support
window.addEventListener('keydown', (e) => {
  const k = e.key;

  if (k >= '0' && k <= '9') {
    e.preventDefault();
    inputDigit(k);
    return;
  }

  if (k === '.') {
    e.preventDefault();
    inputDot();
    return;
  }

  if (k === 'Enter' || k === '=') {
    e.preventDefault();
    equals();
    return;
  }

  if (k === 'Backspace') {
    e.preventDefault();
    backspace();
    return;
  }

  if (k === 'Escape') {
    e.preventDefault();
    clearAll();
    return;
  }

  if (k === '+' || k === '-' || k === '*' || k === '/') {
    e.preventDefault();
    chooseOp(k);
    return;
  }

  if (k === 'x' || k === 'X') {
    e.preventDefault();
    chooseOp('*');
    return;
  }
});

// Init
applyLanguage(langEl.value || 'en');
clearAll();
